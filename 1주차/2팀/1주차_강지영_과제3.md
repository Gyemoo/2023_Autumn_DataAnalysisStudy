나타난 데이터를 사용하여 1월부터 6월까지의 서울의 평균 기온 계산하기


## 답

import pandas as pd
import matplotlib.pyplot as plt

data = pd.DataFrame({
    "월": ["1월", "2월", "3월", "4월", "5월", "6월"],
    "평균 기온": [2, 4, 10, 15, 20, 25]
})

average_temperature = data["평균 기온"].mean()

print("1월부터 6월까지의 서울의 평균 기온은", average_temperature, "°C")
